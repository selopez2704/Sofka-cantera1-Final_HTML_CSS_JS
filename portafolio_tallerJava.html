La Secretaría de Salud Municipal requiere de una aplicación que le permita calcular el IMC (Índice de masa corporal) y requiere los datos peso en kilogramos y estatura en metros Para cada persona encuestada adicional a los datos suministrados, debe mostrar el resultado para cada uno y establecer en qué rango se encuentra (bajo peso, normal, sobrepeso y obeso).

            package ejercicio7_taller2;

            import java.util.Scanner;
            import java.lang.Math;
            public class Main {
                private static Double peso;
                private static Double estatura;
                private static Double imc;
            
            
                public static void main(String[] args) {
                    Scanner read = new Scanner(System.in);
                    System.out.println("Ingrese su peso en kilogramos (ejm: 71,1, etc)");
                    peso=read.nextDouble();
                    System.out.println("Ingrese su estatura en metros (ejm: 1,90)");
                    estatura=read.nextDouble();
                    imc=peso/(Math.pow(estatura,2));
                    if (imc< 18.5){
                        System.out.println("Su rango es BAJO PESO");
                    }else if(imc >= 18.5 && imc <= 24.9){
                        System.out.println("Su rango es NORMAL");
                    } else if (imc > 24.9 && imc <= 29.9) {
                        System.out.println("Su rango es SOBREPESO");
                    }else if(imc > 29.9){
                        System.out.println("Su rango es OBESO");
                    }
                }
            }
