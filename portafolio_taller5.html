            SubProceso area(abc,band)
                definir resul Como real;
                si band=1 Entonces
                    resul<-abc[0]*abc[1];
                FinSi
                si band=2 Entonces
                    resul<-(abc[0]*abc[1])/2;
                FinSi
                si band=3 Entonces
                    resul<-((abc[0]+abc[1])/2)*abc[2];
                FinSi
                Escribir "El area es ", resul;
            FinSubProceso
