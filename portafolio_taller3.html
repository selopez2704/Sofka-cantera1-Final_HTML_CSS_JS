1. Realizar la siguiente figura por medio del ciclo Para

2. Realizar la siguiente figura por medio del ciclo Mientras - Hacer

                Proceso Ejercicio2_taller3_semana1_cantera1
                Definir i, j Como Entero;
                Definir Aste, Esp, print Como Caracter;
                i<-1;
                Mientras i<=10 Hacer
                Esp<-"";
                Aste<-"";
                j<-1;
                Mientras j<=(10-i) Hacer
                Esp<-concatenar(esp," ");
                j<-j+1;
                FinMientras
                j<-1;
                Mientras j<=i Hacer
                Aste<-concatenar(Aste,"*");
                j<-j+1;
                FinMientras;
                print <- concatenar(Esp,Aste);
                Escribir print;
                i<-i+1;
                FinMientras
                FinProceso
